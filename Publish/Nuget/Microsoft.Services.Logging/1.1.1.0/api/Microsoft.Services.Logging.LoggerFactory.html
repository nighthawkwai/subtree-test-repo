Remarks

This class autogenerates loggers for using System.Diagnostics.Tracing.EventSource.

CreateLogger<TLogger>(String) takes as argument an input interface. This is to be annotated with 

   //This is the ETW provider name
   [EtwEventSourceName("Microsoft-HdInsight")]
   [EtwEventNameSuffix("Entry")]
   [EtwEventValueProvider(typeof(TraceLoggerValueProvider))]
   public interface ITraceLogger
   {
       [EtwEventNamePrefix("Log")]
       //This is the prefix of the event name
       //The suffix is defined globally for the interface
       //So the actual etw event name would be "LogEntry"
       void TraceLog(LogMessage message);
   }

   // Public properties defined in the value provider will be added to every log event
   [EventData]
   public class TraceLoggerValueProvider : CallContextLoggingValueProviderBase<TraceLoggerValueProvider>
   {
       public string CorrelationId { get; set; }
   }
